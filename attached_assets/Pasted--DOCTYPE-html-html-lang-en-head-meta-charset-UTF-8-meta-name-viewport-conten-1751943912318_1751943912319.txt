<script>
    let currentSort = { column: -1, direction: 'asc' };
    let selectedPath = 'beginner';
    let resourceData = [];
    
    // Initialize the application
    document.addEventListener('DOMContentLoaded', function() {
        initializeData();
        updateDashboard();
        generateTimeline();
        loadSavedProgress();
    });
    
    function initializeData() {
        const tableRows = document.querySelectorAll('#tableBody tr');
        resourceData = Array.from(tableRows).map((row, index) => {
            const cells = row.querySelectorAll('td');
            return {
                id: index,
                title: cells[0].textContent,
                category: cells[1].textContent,
                type: cells[2].textContent,
                provider: cells[3].textContent,
                duration: cells[4].textContent,
                cost: cells[5].textContent,
                difficulty: cells[6].textContent,
                priority: cells[7].textContent,
                month: cells[8].textContent,
                experienceLevel: cells[9].textContent,
                progress: 'Not Started',
                notes: ''
            };
        });
    }
    
    function showTab(tabName) {
        // Hide all tabs
        document.querySelectorAll('.tab-content').forEach(tab => {
            tab.classList.remove('active');
        });
        
        // Show selected tab
        document.getElementById(tabName).classList.add('active');
        
        // Update nav buttons
        document.querySelectorAll('.nav-tab').forEach(btn => {
            btn.classList.remove('active');
        });
        event.target.classList.add('active');
        
        if (tabName === 'timeline') {
            generateTimeline();
        }
    }
    
    function updateProgress(selectElement) {
        const row = selectElement.closest('tr');
        const resourceId = Array.from(document.querySelectorAll('#tableBody tr')).indexOf(row);
        const newProgress = selectElement.value;
        
        // Update visual style
        selectElement.className = `progress-selector progress-${newProgress.toLowerCase().replace(' ', '-')}`;
        
        // Update data
        if (resourceData[resourceId]) {
            resourceData[resourceId].progress = newProgress;
        }
        
        updateDashboard();
        saveProgress();
    }
    
    function updateDashboard() {
        const total = resourceData.length;
        const completed = resourceData.filter(r => r.progress === 'Completed').length;
        const inProgress = resourceData.filter(r => r.progress === 'In Progress').length;
        const notStarted = resourceData.filter(r => r.progress === 'Not Started').length;
        const progressPercentage = Math.round((completed / total) * 100);
        
        document.getElementById('totalResources').textContent = total;
        document.getElementById('completedResources').textContent = completed;
        document.getElementById('inProgressResources').textContent = inProgress;
        document.getElementById('notStartedResources').textContent = notStarted;
        
        // Update free resources count
        const freeCount = resourceData.filter(r => r.cost && r.cost.includes('Free')).length;
        document.getElementById('freeResources').textContent = freeCount;
        
        // Update advanced resources count
        const advancedCount = resourceData.filter(r => r.difficulty && r.difficulty.includes('Advanced')).length;
        document.getElementById('advancedResources').textContent = advancedCount;
    }
    
    function generateTimeline() {
        const timeline = document.querySelector('.timeline');
        timeline.innerHTML = '';
        
        const monthData = {};
        
        // Group resources by month
        resourceData.forEach(resource => {
            const monthMatch = resource.month.match(/Month (\d+)/);
            if (monthMatch) {
                const monthNum = parseInt(monthMatch[1]);
                if (!monthData[monthNum]) {
                    monthData[monthNum] = [];
                }
                monthData[monthNum].push(resource);
            }
        });
        
        // Create timeline entries
        Object.keys(monthData).sort((a, b) => parseInt(a) - parseInt(b)).forEach(month => {
            const monthDiv = document.createElement('div');
            monthDiv.className = 'timeline-month';
            
            const content = document.createElement('div');
            content.className = 'timeline-content';
            
            const marker = document.createElement('div');
            marker.className = 'timeline-marker';
            
            const title = document.createElement('h3');
            title.className = 'month-title';
            title.textContent = `Month ${month}`;
            
            const resourcesList = document.createElement('ul');
            resourcesList.className = 'month-resources';
            resourcesList.style.listStyle = 'none';
            resourcesList.style.padding = '0';
            
            monthData[month].forEach(resource => {
                const li = document.createElement('li');
                li.style.padding = '8px 0';
                li.style.borderBottom = '1px solid #ecf0f1';
                li.style.display = 'flex';
                li.style.justifyContent = 'space-between';
                li.style.alignItems = 'center';
                
                const resourceInfo = document.createElement('div');
                resourceInfo.className = 'resource-info';
                resourceInfo.innerHTML = `
                    <strong>${resource.title}</strong>
                    <span class="resource-duration" style="font-size: 0.8em; color: #7f8c8d; margin-left: 10px;">${resource.duration}</span>
                `;
                li.appendChild(resourceInfo);
                resourcesList.appendChild(li);
            });
            
            content.appendChild(title);
            content.appendChild(resourcesList);
            monthDiv.appendChild(content);
            monthDiv.appendChild(marker);
            timeline.appendChild(monthDiv);
        });
    }
    
    function sortTable(columnIndex) {
        const table = document.getElementById('resourceTable');
        const tbody = table.querySelector('tbody');
        const rows = Array.from(tbody.querySelectorAll('tr'));
        
        if (currentSort.column === columnIndex) {
            currentSort.direction = currentSort.direction === 'asc' ? 'desc' : 'asc';
        } else {
            currentSort.direction = 'asc';
        }
        currentSort.column = columnIndex;
        
        rows.sort((a, b) => {
            const aVal = a.cells[columnIndex].textContent.trim();
            const bVal = b.cells[columnIndex].textContent.trim();
            
            if (currentSort.direction === 'asc') {
                return aVal.localeCompare(bVal);
            } else {
                return bVal.localeCompare(aVal);
            }
        });
        
        rows.forEach(row => tbody.appendChild(row));
        
        // Update header indicators
        document.querySelectorAll('th').forEach((th, index) => {
            const indicator = th.querySelector('.sort-indicator');
            if (indicator) indicator.remove();
            
            if (index === columnIndex) {
                const arrow = document.createElement('span');
                arrow.className = 'sort-indicator';
                arrow.textContent = currentSort.direction === 'asc' ? ' ‚Üë' : ' ‚Üì';
                arrow.style.position = 'absolute';
                arrow.style.right = '10px';
                th.appendChild(arrow);
            }
        });
    }
    
    function filterTable() {
        const category = document.getElementById('categoryFilter').value;
        const type = document.getElementById('typeFilter').value;
        const difficulty = document.getElementById('difficultyFilter').value;
        const cost = document.getElementById('costFilter').value;
        const progress = document.getElementById('progressFilter').value;
        const month = document.getElementById('monthFilter').value;
        const search = document.getElementById('searchInput').value.toLowerCase();
        
        const tbody = document.getElementById('tableBody');
        const rows = tbody.querySelectorAll('tr');
        
        rows.forEach(row => {
            const cells = row.querySelectorAll('td, select, textarea');
            const title = cells[0].textContent.toLowerCase();
            const rowCategory = cells[1].textContent;
            const rowType = cells[2].textContent;
            const rowDifficulty = cells[6].textContent;
            const rowCost = cells[5].textContent.includes('Free') ? 'Free' : 'Paid';
            const rowProgress = cells[10].value || 'Not Started';
            const rowMonth = cells[8].textContent.match(/\d+/)?.[0];
            
            let show = true;
            
            if (category && !rowCategory.includes(category)) show = false;
            if (type && !rowType.includes(type)) show = false;
            if (difficulty && !rowDifficulty.includes(difficulty)) show = false;
            if (cost && rowCost !== cost) show = false;
            if (progress && rowProgress !== progress) show = false;
            if (month && rowMonth !== month) show = false;
            if (search && !title.includes(search)) show = false;
            
            row.style.display = show ? '' : 'none';
        });
    }
    
    function saveProgress() {
        const progressData = {
            resources: resourceData,
            lastUpdated: new Date().toISOString()
        };
        localStorage.setItem('aiEngineerProgress', JSON.stringify(progressData));
    }
    
    function loadProgress() {
        const saved = localStorage.getItem('aiEngineerProgress');
        if (saved) {
            const data = JSON.parse(saved);
            resourceData = data.resources;
            
            // Update UI
            document.querySelectorAll('#tableBody tr').forEach((row, index) => {
                if (resourceData[index]) {
                    const progressSelect = row.querySelector('.progress-selector');
                    const notesTextarea = row.querySelector('.notes-input');
                    
                    progressSelect.value = resourceData[index].progress;
                    progressSelect.className = `progress-selector progress-${resourceData[index].progress.toLowerCase().replace(' ', '-')}`;
                    notesTextarea.value = resourceData[index].notes || '';
                }
            });
            
            updateDashboard();
        }
    }
    
    function loadSavedProgress() {
        loadProgress();
    }
    
    function resetProgress() {
        if (confirm('Are you sure you want to reset all progress? This cannot be undone.')) {
            localStorage.removeItem('aiEngineerProgress');
            location.reload();
        }
    }
    
    function exportToCSV() {
        const headers = ['Title', 'Category', 'Type', 'Provider', 'Duration', 'Cost', 'Difficulty', 'Priority', 'Month', 'Experience Level', 'Progress', 'Link', 'Personal Notes'];
        
        let csv = [headers.join(',')];
        
        resourceData.forEach(resource => {
            const row = [
                `"${resource.title}"`,
                `"${resource.category}"`,
                `"${resource.type}"`,
                `"${resource.provider}"`,
                `"${resource.duration}"`,
                `"${resource.cost}"`,
                `"${resource.difficulty}"`,
                `"${resource.priority}"`,
                `"${resource.month}"`,
                `"${resource.experienceLevel}"`,
                `"${resource.progress}"`,
                `"${resource.link || ''}"`,
                `"${resource.notes || ''}"`
            ];
            csv.push(row.join(','));
        });
        
        const csvContent = csv.join('\n');
        const blob = new Blob([csvContent], { type: 'text/csv' });
        const url = window.URL.createObjectURL(blob);
        const a = document.createElement('a');
        a.href = url;
